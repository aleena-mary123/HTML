1)Javascript program for prime number.

<script>
function test_prime(n)
{

  if (n===1)
  {
    return false;
  }
  else if(n === 2)
  {
    return true;
  }else
  {
    for(var x = 2; x < n; x++)
    {
      if(n % x === 0)
      {
        return false;
      }
    }
    return true;  
  }
}

console.log(test_prime(37));
</script>

4)Javascript program to swap 2 variables.

<script>
	var a = 20;
	var b = 10;
	var temp;

	console.log(`before swapping: a= ${a}`);
	console.log(`before swapping b= ${b}`);

	temp = a;
	a = b;
	b = temp;

	console.log(`after swapping a= ${a}`);
	console.log(`after swapping b= ${b}`);
</script>
